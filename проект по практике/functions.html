Функции в JavaScript

Функции позволяют организовать код в удобные для повторного использования блоки.

Объявление функции

Функции можно объявлять разными способами.

// Function Declaration (можно вызывать до объявления)
function sayHello() {
    console.log("Привет, мир!");
}

sayHello(); // "Привет, мир!"

Функциональные выражения

Функция может быть записана в переменную.

const sayHello = function() {
    console.log("Привет!");
};

sayHello();

Стрелочные функции

Более короткий синтаксис, без this и arguments.

const sum = (a, b) => a + b;
console.log(sum(5, 3)); // 8

Параметры по умолчанию

Параметры функции могут иметь значения по умолчанию.

function greet(name = "Гость") {
    console.log("Привет, " + name);
}

greet(); // "Привет, Гость"
greet("Алексей"); // "Привет, Алексей"

Rest-параметры

Позволяют передавать переменное количество аргументов.

function sum(...numbers) {
    return numbers.reduce((acc, num) => acc + num, 0);
}

console.log(sum(1, 2, 3, 4)); // 10

Область видимости

Функции имеют свою область видимости.

function outer() {
    let outerVar = "Я снаружи";

    function inner() {
        console.log(outerVar); // "Я снаружи"
    }

    inner();
}

outer();

Замыкания

Функция, запоминающая свою область видимости.

function makeCounter() {
    let count = 0;

    return function() {
        return ++count;
    };
}

const counter = makeCounter();
console.log(counter()); // 1
console.log(counter()); // 2

Колбэк-функции

Функции, передаваемые как аргументы.

function greet(name, callback) {
    console.log("Привет, " + name);
    callback();
}

greet("Алекс", function() {
    console.log("Как дела?");
});

Методы объектов

Функции внутри объектов называются методами.

let user = {
    name: "Алекс",
    sayHi() {
        console.log("Привет, меня зовут " + this.name);
    }
};

user.sayHi(); // "Привет, меня зовут Алекс"

Функции-конструкторы

Используются для создания объектов.

function User(name, age) {
    this.name = name;
    this.age = age;
}

let user1 = new User("Иван", 30);
console.log(user1.name); // "Иван"

Классы и методы

Синтаксический сахар над конструкторами.

class User {
    constructor(name, age) {
        this.name = name;
        this.age = age;
    }

    sayHi() {
        console.log("Привет, я " + this.name);
    }
}

let user2 = new User("Ольга", 25);
user2.sayHi(); // "Привет, я Ольга"

Геттеры и сеттеры

Позволяют управлять доступом к свойствам.

class User {
    constructor(name) {
        this._name = name;
    }

    get name() {
        return this._name.toUpperCase();
    }

    set name(value) {
        if (value.length < 3) {
            console.log("Имя слишком короткое!");
            return;
        }
        this._name = value;
    }
}

let user3 = new User("Иван");
console.log(user3.name); // "ИВАН"
user3.name = "Al"; // Ошибка

Рекурсия

Функция вызывает саму себя.

function factorial(n) {
    return n === 1 ? 1 : n * factorial(n - 1);
}

console.log(factorial(5)); // 120

Функции высшего порядка

Функции, принимающие другие функции в качестве аргументов.

function applyOperation(a, b, operation) {
    return operation(a, b);
}

console.log(applyOperation(5, 3, (x, y) => x + y)); // 8
